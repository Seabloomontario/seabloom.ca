    <script>
        function toggleMobileMenu() {
            const menu = document.getElementById('mobile-menu');
            menu.classList.toggle('hidden');
        }

        function scrollToContact() {
            document.getElementById('contact').scrollIntoView({ behavior: 'smooth' });
        }

        function toggleReviewModal() {
            // Simplified for the single-page experience
            alert("This feature would allow clients to submit live reviews.");
        }

        function handleContact(e) {
            e.preventDefault();
            const button = e.target.querySelector('button');
            const successMsg = document.getElementById('success-message');
            
            button.innerHTML = '<i class="fa-solid fa-spinner fa-spin mr-2"></i> Sending...';
            button.disabled = true;
            
            setTimeout(() => {
                button.innerHTML = 'Request Received';
                button.classList.replace('bg-primary', 'bg-green-600');
                successMsg.classList.remove('hidden');
                e.target.reset();
                
                setTimeout(() => {
                    button.innerHTML = 'Send My Request';
                    button.classList.replace('bg-green-600', 'bg-primary');
                    button.disabled = false;
                    successMsg.classList.add('hidden');
                }, 5000);
            }, 1500);
        }
